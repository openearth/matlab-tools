DESCRIPTION 

 KDEDEMO1 Demonstrate the smoothing parameter impact on KDE 
  
  KDEDEMO1 shows the true density (dotted) compared to KDE based on 7 
  observations (solid) and their individual kernels (dashed) for 3 
  different values of the smoothing parameter, hs.

023  data   = wnormrnd(0,1,7,1);

025  hs     = hns(data);

029  kopt = kdeoptset('kernel','gaus');

032    h = hVec(ix); 
033    kopt.hs = h; 
034    f2 = kde(data,kopt,x); 
035    pdfplot(f2,'k-') 
036    title(sprintf('h_s = %s', num2str(h,2))) 
037    ylabel('Density') 
038     
039    hold on 
040    plot(x,wnormpdf(x,0,1),'k:') 
041    n = length(data); 
042    plot(data,zeros(size(data)),'bx','Markersize',10) 
043    y = mkernel(x0,kernel)/(n*h);  
044    for i=1:n 
045      plot(data(i)+x0*h,y,'b--'); 
046      plot([data(i) data(i)], [0 max(y)],'b') 
047    end 
048    set(gca,'ytick',0:.1:0.5)
